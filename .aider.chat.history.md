# -*- coding: utf-8 -*-
"""
Main scraper module using Playwright.
Handles browser automation, navigation, and data extraction.
"""

import asyncio
import json
import re
from typing import List, Dict, Optional
from playwright.async_api import async_playwright, Page, Browser
from loguru import logger

from .config import BASE_URLS, SELECTORS, SCRAPER_CONFIG


class TokopediaScraper:
    """
    Scraper class for extracting laptop data from Tokopedia.
    Uses Playwright for async browser automation.
    """

    def __init__(self):
        self.results: List[Dict] = []
        self.browser: Optional[Browser] = None

    async def init_browser(self):
        """Initialize Playwright browser instance."""
        self.playwright = await async_playwright().start()
        self.browser = await self.playwright.chromium.launch(
            headless=SCRAPER_CONFIG["headless"]
        )
        logger.info("Browser initialized successfully")

    async def close_browser(self):
        """Close the browser instance."""
        if self.browser:
            await self.browser.close()
        if self.playwright:
            await self.playwright.stop()
        logger.info("Browser closed")

    async def scroll_page(self, page: Page):
        """
        Scroll down the page to trigger lazy loading.
        Tokopedia uses infinite scroll on search pages.
        """
        last_height = 0
        scroll_attempts = 0
        max_scrolls = 10  # Limit scrolls to prevent infinite loops

        while scroll_attempts < max_scrolls:
            await page.evaluate("window.scrollTo(0, document.body.scrollHeight)")
            await asyncio.sleep(SCRAPER_CONFIG["delay_between_requests"])
            
            new_height = await page.evaluate("document.body.scrollHeight")
            if new_height == last_height:
                break
            last_height = new_height
            scroll_attempts += 1
        
        logger.debug(f"Scrolled {scroll_attempts} times")

    async def extract_product_data(self, page: Page) -> List[Dict]:
        """
        Extract product information from the current page.
        Returns a list of product dictionaries.
        """
        products = []
        
        # Wait for products to load
        try:
            await page.wait_for_selector(SELECTORS["product_card"], timeout=SCRAPER_CONFIG["page_timeout"])
        except Exception as e:
            logger.warning(f"Timeout waiting for products: {e}")
            return products

        # Get all product cards
        product_cards = await page.query_selector_all(SELECTORS["product_card"])
        logger.info(f"Found {len(product_cards)} product cards")

        for card in product_cards:
            try:
                product_data = await self._parse_product_card(card)
                if product_data:
                    products.append(product_data)
            except Exception as e:
                logger.error(f"Error parsing product card: {e}")
                continue

        return products

    async def _parse_product_card(self, card) -> Optional[Dict]:
        """
        Parse individual product card HTML element.
        """
        try:
            # Extract name
            name_elem = await card.query_selector(SELECTORS["product_name"])
            name = await name_elem.inner_text() if name_elem else ""
            
            # Extract price
            price_elem = await card.query_selector(SELECTORS["product_price"])
            price_text = await price_elem.inner_text() if price_elem else "0"
            price = self._parse_price(price_text)
            
            # Extract link
            link_elem = await card.query_selector(SELECTORS["product_link"])
            link = await link_elem.get_attribute("href") if link_elem else ""
            
            # Extract rating
            rating_elem = await card.query_selector(SELECTORS["product_rating"])
            rating_text = await rating_elem.inner_text() if rating_elem else "0"
            rating = self._parse_rating(rating_text)
            
            # Extract image
            img_elem = await card.query_selector(SELECTORS["product_image"])
            image = await img_elem.get_attribute("src") if img_elem else ""

            if not name or price == 0:
                return None

            return {
                "name": name.strip(),
                "price": price,
                "rating": rating,
                "link": link,
                "image": image,
                "source": "tokopedia"
            }
        except Exception as e:
            logger.debug(f"Skipping card due to error: {e}")
            return None

    def _parse_price(self, price_text: str) -> int:
        """
        Parse price string to integer.
        Example: 'Rp10.500.000' -> 10500000
        """
        # Remove 'Rp', spaces, and dots
        cleaned = re.sub(r'[Rp\s.]', '', price_text)
        try:
            return int(cleaned)
        except ValueError:
            return 0

    def _parse_rating(self, rating_text: str) -> float:
        """
        Parse rating string to float.
        Example: '4.8' -> 4.8
        """
        try:
            return float(rating_text.split()[0])
        except (ValueError, IndexError):
            return 0.0

    async def scrape_brand(self, url: str) -> List[Dict]:
        """
        Scrape products from a specific brand URL.
        """
        page = await self.browser.new_page()
        products = []
        
        try:
            logger.info(f"Navigating to: {url}")
            await page.goto(url, timeout=SCRAPER_CONFIG["timeout"])
            
            # Accept cookies if popup appears
            try:
                accept_btn = await page.query_selector('button:has-text("Terima")')
                if accept_btn:
                    await accept_btn.click()
                    await asyncio.sleep(1)
            except Exception:
                pass

            # Scroll to load all products
            await self.scroll_page(page)
            
            # Extract data
            products = await self.extract_product_data(page)
            logger.info(f"Extracted {len(products)} products from {url}")

        except Exception as e:
            logger.error(f"Error scraping {url}: {e}")
        finally:
            await page.close()

        return products

    async def run(self) -> List[Dict]:
        """
        Main entry point. Scrapes all configured URLs.
        """
        await self.init_browser()
        all_products = []

        try:
            for url in BASE_URLS:
                products = await self.scrape_brand(url)
                all_products.extend(products)
                
                # Rate limiting
                await asyncio.sleep(SCRAPER_CONFIG["delay_between_requests"])

            # Deduplicate by name
            seen = set()
            unique_products = []
            for p in all_products:
                if p["name"] not in seen:
                    seen.add(p["name"])
                    unique_products.append(p)

            logger.info(f"Total unique products scraped: {len(unique_products)}")
            self.results = unique_products
            return unique_products

        finally:
            await self.close_browser()


async def main():
    """Standalone execution for testing."""
    scraper = TokopediaScraper()
    results = await scraper.run()
    
    # Save to JSON for inspection
    with open("scraped_data.json", "w", encoding="utf-8") as f:
        json.dump(results, f, ensure_ascii=False, indent=2)
    
    print(f"Scraped {len(results)} products. Saved to scraped_data.json")


if __name__ == "__main__":
    asyncio.run(main())

# -*- coding: utf-8 -*-
"""
Parser module for extracting laptop specifications from product names.
Uses regex patterns to identify CPU, RAM, Storage, etc.
"""

import re
from typing import Dict, Optional, List
from loguru import logger


class SpecParser:
    """
    Parses laptop specifications from product titles and descriptions.
    Indonesian e-commerce titles often follow specific patterns.
    """

    # Brand patterns
    BRANDS = {
        "ASUS": r"\b(asus)\b",
        "Lenovo": r"\b(lenovo)\b",
        "HP": r"\b(hp|hp)\b",
        "Acer": r"\b(acer)\b",
        "Apple": r"\b(apple|macbook|mba|mbp)\b",
        "Dell": r"\b(dell)\b",
        "MSI": r"\b(msi)\b",
        "LG": r"\b(lg)\b",
    }

    # CPU patterns
    CPU_INTEL = r"\b(i[3579][- ]?\d{4,5}[a-z]*)\b|\b(core\s*i[3579])\b"
    CPU_AMD = r"\b(ryzen\s*\d\s*\d{4})\b|\b(r[3579][- ]?\d{4})\b"
    CPU_APPLE = r"\b(m[123]\s*(pro|max|ultra)?)\b|\b(apple\s*silicon)\b"

    # RAM patterns (e.g., 8GB, 16GB, 32GB)
    RAM = r"\b(\d+)\s*gb\s*(ram|ddr|lpddr)?\b"

    # Storage patterns (e.g., 512GB SSD, 1TB)
    STORAGE = r"\b(\d+)\s*(gb|tb)\s*(ssd|nvme|sata)?\b"

    # Screen size (e.g., 14", 15.6 inch)
    SCREEN = r"\b(\d{2}(?:\.\d)?)\s*(inch|\"|in)\b"

    # GPU patterns
    GPU_DEDICATED = r"\b(gtx\s*\d+|rtx\s*\d+|mx\s*\d+|gtx|rtx)\b"

    def parse_title(self, title: str) -> Dict:
        """
        Main parsing function. Extracts all specs from a product title.
        """
        title_lower = title.lower()
        
        specs = {
            "brand": self._extract_brand(title_lower),
            "cpu_model": self._extract_cpu(title_lower),
            "ram_gb": self._extract_ram(title_lower),
            "storage_gb": self._extract_storage(title_lower),
            "screen_inches": self._extract_screen(title_lower),
            "gpu_model": self._extract_gpu(title_lower),
            "raw_title": title
        }

        return specs

    def _extract_brand(self, text: str) -> Optional[str]:
        """Extract brand name from text."""
        for brand, pattern in self.BRANDS.items():
            if re.search(pattern, text, re.IGNORECASE):
                return brand
        return None

    def _extract_cpu(self, text: str) -> Optional[str]:
        """Extract CPU model from text."""
        # Try Intel
        intel_match = re.search(self.CPU_INTEL, text, re.IGNORECASE)
        if intel_match:
            return intel_match.group(1).upper().replace(" ", "-")
        
        # Try AMD
        amd_match = re.search(self.CPU_AMD, text, re.IGNORECASE)
        if amd_match:
            return amd_match.group(1).upper().replace(" ", "-")
        
        # Try Apple
        apple_match = re.search(self.CPU_APPLE, text, re.IGNORECASE)
        if apple_match:
            return apple_match.group(1).upper()

        return None

    def _extract_ram(self, text: str) -> Optional[int]:
        """Extract RAM size in GB."""
        match = re.search(self.RAM, text, re.IGNORECASE)
        if match:
            return int(match.group(1))
        return None

    def _extract_storage(self, text: str) -> Optional[int]:
        """Extract storage size in GB."""
        matches = re.findall(self.STORAGE, text, re.IGNORECASE)
        if matches:
            # Get the largest storage value found
            max_storage = 0
            for match in matches:
                size = int(match[0])
                unit = match[1].lower()
                if unit == 'tb':
                    size *= 1000
                if size > max_storage:
                    max_storage = size
            return max_storage
        return None

    def _extract_screen(self, text: str) -> Optional[float]:
        """Extract screen size in inches."""
        match = re.search(self.SCREEN, text, re.IGNORECASE)
        if match:
            return float(match.group(1))
        return None

    def _extract_gpu(self, text: str) -> Optional[str]:
        """Extract dedicated GPU model if present."""
        match = re.search(self.GPU_DEDICATED, text, re.IGNORECASE)
        if match:
            return match.group(1).upper()
        return None

    def generate_slug(self, brand: str, title: str) -> str:
        """
        Generate a URL-friendly slug for the laptop.
        Example: 'ASUS VivoBook 14 X1404VA' -> 'asus-vivobook-14-x1404va'
        """
        # Remove common noise words
        noise_words = ['laptop', 'notebook', 'original', 'resmi', 'garansi', 'new', 'seal']
        clean_title = title.lower()
        for word in noise_words:
            clean_title = re.sub(rf'\b{word}\b', '', clean_title)
        
        # Remove special characters
        clean_title = re.sub(r'[^a-z0-9\s-]', '', clean_title)
        
        # Create slug
        slug_parts = [brand.lower()] + clean_title.split()
        slug = '-'.join(slug_parts)
        slug = re.sub(r'-+', '-', slug)  # Remove duplicate dashes
        slug = slug.strip('-')
        
        return slug[:100]  # Limit length


def parse_products(raw_products: List[Dict]) -> List[Dict]:
    """
    Batch parse multiple products.
    """
    parser = SpecParser()
    parsed = []

    for product in raw_products:
        try:
            specs = parser.parse_title(product.get("name", ""))
            
            parsed_product = {
                **product,
                **specs,
                "slug": parser.generate_slug(specs.get("brand", ""), product.get("name", ""))
            }
            parsed.append(parsed_product)
            
        except Exception as e:
            logger.error(f"Error parsing product {product.get('name')}: {e}")

    logger.info(f"Parsed {len(parsed)} products successfully")
    return parsed

# -*- coding: utf-8 -*-
"""
Data cleaning and normalization module.
Ensures data consistency before database insertion.
"""

import re
from typing import Dict, List, Optional
from loguru import logger


class DataCleaner:
    """
    Cleans and normalizes laptop data.
    """

    # Known brand mappings (for normalization)
    BRAND_MAPPING = {
        "asus": "ASUS",
        "lenovo": "Lenovo",
        "hp": "HP",
        "acer": "Acer",
        "apple": "Apple",
        "dell": "Dell",
        "msi": "MSI",
        "lg": "LG",
    }

    # CPU brand mapping
    CPU_BRANDS = {
        "intel": ["i3", "i5", "i7", "i9", "core"],
        "amd": ["ryzen", "r3", "r5", "r7", "r9"],
        "apple": ["m1", "m2", "m3", "m4", "silicon"]
    }

    def clean_product(self, product: Dict) -> Dict:
        """
        Clean a single product dictionary.
        """
        cleaned = product.copy()

        # Normalize brand
        cleaned["brand"] = self._normalize_brand(product.get("brand"))

        # Normalize RAM
        cleaned["ram_gb"] = self._normalize_number(product.get("ram_gb"))

        # Normalize Storage
        cleaned["storage_gb"] = self._normalize_number(product.get("storage_gb"))

        # Normalize Screen Size
        cleaned["screen_inches"] = self._normalize_float(product.get("screen_inches"))

        # Clean price (remove invalid values)
        if product.get("price", 0) <= 0:
            logger.warning(f"Invalid price for {product.get('name')}")
            cleaned["price"] = None

        # Ensure slug exists
        if not product.get("slug"):
            cleaned["slug"] = self._generate_fallback_slug(product.get("name", ""))

        # Determine GPU type
        cleaned["gpu_type"] = "dedicated" if product.get("gpu_model") else "integrated"

        return cleaned

    def _normalize_brand(self, brand: Optional[str]) -> str:
        """Normalize brand name to standard format."""
        if not brand:
            return "Unknown"
        
        brand_lower = brand.lower().strip()
        return self.BRAND_MAPPING.get(brand_lower, brand.title())

    def _normalize_number(self, value: Optional[int]) -> Optional[int]:
        """Ensure numeric values are valid."""
        if value is None:
            return None
        try:
            return int(value)
        except (ValueError, TypeError):
            return None

    def _normalize_float(self, value: Optional[float]) -> Optional[float]:
        """Ensure float values are valid."""
        if value is None:
            return None
        try:
            return float(value)
        except (ValueError, TypeError):
            return None

    def _generate_fallback_slug(self, name: str) -> str:
        """Generate a slug if parsing failed."""
        slug = name.lower()
        slug = re.sub(r'[^a-z0-9]+', '-', slug)
        slug = slug.strip('-')
        return slug[:100]

    def validate_for_db(self, product: Dict) -> bool:
        """
        Check if product has minimum required fields for database insertion.
        """
        required = ["name", "slug", "brand", "price"]
        
        for field in required:
            if not product.get(field):
                logger.warning(f"Missing required field '{field}' for {product.get('name', 'Unknown')}")
                return False
        
        return True

    def clean_batch(self, products: List[Dict]) -> List[Dict]:
        """
        Clean a batch of products and filter out invalid ones.
        """
        cleaned = []
        
        for product in products:
            cleaned_product = self.clean_product(product)
            if self.validate_for_db(cleaned_product):
                cleaned.append(cleaned_product)
        
        logger.info(f"Cleaned {len(cleaned)} products (filtered from {len(products)})")
        return cleaned

# -*- coding: utf-8 -*-
"""
Database synchronization module.
Handles upserting laptop data to Supabase.
"""

import os
from typing import List, Dict, Optional
from loguru import logger

try:
    from supabase import create_client, Client
except ImportError:
    logger.warning("supabase package not installed. Run: pip install supabase")
    Client = None


class DatabaseSync:
    """
    Handles database operations with Supabase.
    """

    def __init__(self):
        self.client: Optional[Client] = None
        self._connect()

    def _connect(self):
        """Initialize Supabase client from environment variables."""
        supabase_url = os.getenv("SUPABASE_URL")
        supabase_key = os.getenv("SUPABASE_SERVICE_KEY") or os.getenv("SUPABASE_ANON_KEY")

        if not supabase_url or not supabase_key:
            logger.error("Missing SUPABASE_URL or SUPABASE_KEY environment variables")
            return

        try:
            self.client = create_client(supabase_url, supabase_key)
            logger.info("Connected to Supabase successfully")
        except Exception as e:
            logger.error(f"Failed to connect to Supabase: {e}")

    def upsert_laptops(self, laptops: List[Dict]) -> int:
        """
        Upsert laptop data to the database.
        Returns the number of successfully inserted/updated records.
        """
        if not self.client:
            logger.error("Database client not initialized")
            return 0

        if not laptops:
            logger.warning("No laptops to upsert")
            return 0

        success_count = 0

        for laptop in laptops:
            try:
                # Prepare data for insertion
                db_record = self._prepare_record(laptop)
                
                # Upsert based on slug
                response = self.client.table("laptops").upsert(
                    db_record,
                    on_conflict="slug"
                ).execute()
                
                success_count += 1
                
            except Exception as e:
                logger.error(f"Error upserting {laptop.get('slug')}: {e}")

        logger.info(f"Successfully upserted {success_count}/{len(laptops)} laptops")
        return success_count

    def _prepare_record(self, laptop: Dict) -> Dict:
        """
        Transform scraper data format to database schema format.
        """
        return {
            "slug": laptop.get("slug"),
            "brand": laptop.get("brand"),
            "model": laptop.get("name"),  # Use name as model initially
            "full_name": laptop.get("name"),
            "price_tokopedia": laptop.get("price"),
            "cpu_model": laptop.get("cpu_model"),
            "ram_gb": laptop.get("ram_gb"),
            "storage_gb": laptop.get("storage_gb"),
            "screen_inches": laptop.get("screen_inches"),
            "gpu_model": laptop.get("gpu_model"),
            "gpu_type": laptop.get("gpu_type"),
            "images": [laptop.get("image")] if laptop.get("image") else [],
            "source_url": laptop.get("link"),
            "rating": laptop.get("rating"),
            "last_scraped_at": "now()"
        }

    def get_existing_slugs(self) -> List[str]:
        """
        Get list of existing laptop slugs from database.
        Useful for avoiding duplicates.
        """
        if not self.client:
            return []

        try:
            response = self.client.table("laptops").select("slug").execute()
            return [item["slug"] for item in response.data]
        except Exception as e:
            logger.error(f"Error fetching existing slugs: {e}")
            return []

    def mark_inactive(self, active_slugs: List[str]) -> int:
        """
        Mark laptops as inactive if they weren't in the latest scrape.
        (Optional: depends on your business logic)
        """
        # This would require an 'is_active' column in the database
        # For now, we just log the difference
        existing = self.get_existing_slugs()
        inactive = set(existing) - set(active_slugs)
        
        if inactive:
            logger.info(f"{len(inactive)} laptops not found in latest scrape")
        
        return len(inactive)

# -*- coding: utf-8 -*-
"""
Main entry point for the scraper.
Orchestrates: Scrape -> Parse -> Clean -> Sync to DB
"""

import asyncio
import json
from datetime import datetime
from loguru import logger

from .tokopedia import TokopediaScraper
from .parser import parse_products
from .cleaners import DataCleaner
from .db_sync import DatabaseSync


def setup_logging():
    """Configure loguru for pretty output."""
    logger.add(
        "logs/scraper_{time}.log",
        rotation="1 day",
        retention="7 days",
        level="INFO"
    )


async def run_scraper():
    """
    Main async function to run the entire scraping pipeline.
    """
    start_time = datetime.now()
    logger.info(f"Starting scraper at {start_time}")

    # Step 1: Scrape raw data
    logger.info("Step 1: Scraping raw data from Tokopedia...")
    scraper = TokopediaScraper()
    raw_products = await scraper.run()

    if not raw_products:
        logger.error("No products scraped. Exiting.")
        return

    # Step 2: Parse specifications
    logger.info("Step 2: Parsing specifications from titles...")
    parsed_products = parse_products(raw_products)

    # Step 3: Clean data
    logger.info("Step 3: Cleaning and normalizing data...")
    cleaner = DataCleaner()
    clean_products = cleaner.clean_batch(parsed_products)

    # Step 4: Sync to database
    logger.info("Step 4: Syncing to Supabase...")
    db = DatabaseSync()
    inserted = db.upsert_laptops(clean_products)

    # Summary
    end_time = datetime.now()
    duration = (end_time - start_time).total_seconds()
    
    logger.info(f"""
    ===== Scraping Complete =====
    Raw products:     {len(raw_products)}
    Parsed products:  {len(parsed_products)}
    Clean products:   {len(clean_products)}
    DB inserted:      {inserted}
    Duration:         {duration:.2f} seconds
    ==============================
    """)

    # Save backup to JSON
    backup_file = f"backup_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
    with open(backup_file, "w", encoding="utf-8") as f:
        json.dump(clean_products, f, ensure_ascii=False, indent=2)
    logger.info(f"Backup saved to {backup_file}")


def main():
    """Synchronous entry point."""
    setup_logging()
    asyncio.run(run_scraper())


if __name__ == "__main__":
    main()
